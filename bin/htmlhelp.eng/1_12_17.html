SYNOPSIS

TString tldntomscid(TString tldn)

DESCRIPTION

To get the MSCID according to the config.mscid configuration

ARGUMENTS

tldn

the TLDN number

RETURN VALUES

The MSCID

EXAMPLES

void main()
{
   var TString tldn,mscid;
   tldn = '13334710';
   mscid = tldntomscid(tldn);
   assert(mscid == '37288A');
   tldn = '8613334710'
   mscid = tldntomscid(tldn);
   assert(mscid == '37288A');
   tldn = '013334710';
   mscid = tldntomscid(tldn);
   assert(mscid == '37288A');
   tldn = '13334710999';
   mscid = tldntomscid(tldn);
   assert(mscid == '37288A');
}
