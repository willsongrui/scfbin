SYNOPSIS

TString getsipnodeaddress(TString arg)

DESCRIPTION

get node's ip according to node name

ARGUMENTS

arg

node name

RETURN VALUES

The ip of node

EXAMPLES

#progma c
void main()
{
   var TString addresshost;
   addresshost =getsipnodeaddress('ms1');
   assert(addresshost=='192.168.2.237:50914');

}
